node {

	def mavenHome = tool name: "maven3.6.3"
  
    echo "GitHub BranhName ${env.BRANCH_NAME}"
    echo "Jenkins Job Number ${env.BUILD_NUMBER}"
    echo "Jenkins Node Name ${env.NODE_NAME}"
  
    echo "Jenkins Home ${env.JENKINS_HOME}"
    echo "Jenkins URL ${env.JENKINS_URL}"
    echo "JOB Name ${env.JOB_NAME}"
	
	stage('Source Code Checkout')
	{
		git branch: 'main', credentialsId: '<>', url: 'https://github.com/chanduej/Multi-Module-Maven-Project.git'
	}
	stage('Build Artifacts')
	{
		sh "${mavenHome}/bin/mvn clean package"
	}
	stage("UploadArtifactsintoNexus")
	{
		sh "${mavenHome}/bin/mvn deploy"
	}
	stage('Docker Image Build')
	{
		sh "docker build -t tiger5559/multimodulemaven:1 ."
	}
	stage('Deploy Container')
	{
		sh "docker run -d -p 8080:8080 tiger5559/multimodulemaven sample_containername"
	}
}
